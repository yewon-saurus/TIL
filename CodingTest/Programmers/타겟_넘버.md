class Solution {
    static int answer = 0;
    static int arr[];
    static int tmp;
    public int solution(int[] numbers, int target) {
        arr = numbers;
        tmp = target;
        dfs(0, 0);

        return answer;
    }

    public void dfs(int idx, int sum) {
        if (idx == arr.length) {
            if (sum == tmp) {
                answer++;
            }
        }
        else {
            dfs(idx+1, sum+arr[idx]);
            dfs(idx+1, sum-arr[idx]);
        }
    }
}
