import java.util.*;

class Solution {
    static int[] dx = {1, 0, -1, 0};
    static int[] dy = {0, 1, 0, -1};
    static boolean visited[][];
    static int arr[][];
    static int N, M;

    public int solution(int[][] maps) {
        int answer = 0;

        arr = maps;
        N = maps.length;
        M = maps[0].length;
        visited = new boolean[N][M];

        arr[N-1][M-1] = -1;
        bfs(0, 0);
        answer = arr[N-1][M-1];

        return answer;
    }

    public void bfs(int n, int m) {
        Queue<int[]> q = new LinkedList<>();
        q.offer(new int[] {n, m});

        while (!q.isEmpty()) {
            int tmp[] = q.poll();
            for (int i = 0; i <4; i++) {
                int x = tmp[0] + dx[i]; // ê¹”ì§ ì‚´í´ë³´ê¸°
                int y = tmp[1] + dy[i];
                if (x >= 0 && y >= 0 && x < N && y < M) {
                    if (arr[x][y] != 0 && !visited[x][y]) {
                        visited[x][y] = true;
                        arr[x][y] = arr[tmp[0]][tmp[1]] + 1;
                        q.offer(new int[] {x, y});
                    }
                }
            }
        }
    }
}
